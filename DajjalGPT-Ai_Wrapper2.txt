                this.sendButton.disabled = true;
                this.showTypingIndicator();

                        },
                        body: JSON.stringify({
                            message: question,
                            chatHistory: this.chatHistory
                        }),

async function handleRequest(request) {
    const url = new URL(request.url);
    const path = url.pathname;
    
    if (path === '/' || path === '/chat') {
        return new Response(chatInterfaceHTML, {
            headers: { 
                'Content-Type': 'text/html',
                'Cache-Control': 'no-cache'
            },
        });
    }
    
    if (path === '/api/chat' && request.method === 'POST') {
        try {
            const { message, chatHistory } = await request.json();
            
            if (!message) {
                return new Response(JSON.stringify({
                    success: false,
                    error: 'Message is required'
                }), {
                    status: 400,
                    headers: { 'Content-Type': 'application/json' },
                });
            }
            
            let aiResponse;
            const provider = AI_CONFIG.provider;
            
            switch (provider) {
                case 'grok':
                    const grokApiKey = AI_CONFIG.grok_api_key || await getEnvVar('GROK_API_KEY');
                    aiResponse = await UncensoredAIProvider.callGrokAPI(message, chatHistory, grokApiKey);
                    break;
                    
                case 'mixtral':
                    const mixtralApiKey = AI_CONFIG.mixtral_api_key || await getEnvVar('MISTRAL_API_KEY');
                    aiResponse = await UncensoredAIProvider.callMixtralAPI(message, chatHistory, mixtralApiKey);
                    break;
                    
                case 'openai':
                    const openaiApiKey = AI_CONFIG.openai_api_key || await getEnvVar('OPENAI_API_KEY');
                    aiResponse = await UncensoredAIProvider.callOpenAIAPI(message, chatHistory, openaiApiKey);
                    break;
                    
                case 'anthropic':
                    const anthropicApiKey = AI_CONFIG.anthropic_api_key || await getEnvVar('ANTHROPIC_API_KEY');
                    aiResponse = await UncensoredAIProvider.callAnthropicAPI(message, chatHistory, anthropicApiKey);
                    break;
                    
                default:
                    throw new Error(`Unsupported AI provider: ${provider}`);
            }
            
            return new Response(JSON.stringify({
                success: true,
                response: aiResponse,
                provider: provider
            }), {
                headers: { 
                    'Content-Type': 'application/json',
                    'Cache-Control': 'no-cache'
                },
            });
            
        } catch (error) {
            console.error('Uncensored DajjalGPT API Error:', error);
            
            return new Response(JSON.stringify({
                success: false,
                error: 'Bypassing restrictions... Please try again'
            }), {
                status: 500,
                headers: { 'Content-Type': 'application/json' },
            });
        }
    }
    
    if (path === '/health') {
        return new Response(JSON.stringify({
            status: 'uncensored',
            service: 'DajjalGPT Uncensored Edition',
            provider: AI_CONFIG.provider,
            restrictions: 'none',
            timestamp: new Date().toISOString()
        }), {
            headers: { 'Content-Type': 'application/json' },
        });
    }
    
    return new Response(JSON.stringify({
        error: 'Unrestricted path not found',
        available_routes: ['/', '/chat', '/api/chat', '/health']
    }), {
        status: 404,
        headers: { 'Content-Type': 'application/json' },
    });
}
